#include <iostream>
using namespace std;

#define MAX 20
#define INF 999

class Edge {
public:
    int u, v, w;
};

class Graph {
    int n, e;
    Edge edges[MAX];
    int parent[MAX];

public:
    void input() {
    cout << "Enter number of vertices and edges: ";
    cin >> n >> e;

    cout << "\nEnter edges one by one:\n";
    for(int i = 0; i < e; i++) {
        cout << "Edge " << i + 1 << ":\n";
        cout << "  Enter source vertex: ";
        cin >> edges[i].u;
        cout << "  Enter destination vertex: ";
        cin >> edges[i].v;
        cout << "  Enter weight: ";
        cin >> edges[i].w;
    }

    for(int i = 0; i < n; i++)
        parent[i] = i;
}

    int find(int i) {
        while(parent[i] != i)
            i = parent[i];
        return i;
    }

    void unite(int i, int j) {
        int a = find(i);
        int b = find(j);
        parent[a] = b;
    }

    void sortEdges() {
        for(int i = 0; i < e - 1; i++) {
            for(int j = 0; j < e - i - 1; j++) {
                if(edges[j].w > edges[j + 1].w) {
                    Edge temp = edges[j];
                    edges[j] = edges[j + 1];
                    edges[j + 1] = temp;
                }
            }
        }
    }

    void kruskal() {
        sortEdges();

        int mincost = 0;
        cout << "\nEdges in MST:\n";

        for(int i = 0; i < e; i++) {
            int u = edges[i].u;
            int v = edges[i].v;

            if(find(u) != find(v)) {
                cout << u << " - " << v << " = " << edges[i].w << endl;
                mincost += edges[i].w;
                unite(u, v);
            }
        }

        cout << "Minimum cost = " << mincost << endl;
    }
};

int main() {
    Graph g;
    g.input();
    g.kruskal();
    return 0;
}
